                                .attr("href", fullTableLink)
                                .show();
                        } else {
                            $footer.children().hide()
                        }

                            var newTable = {
                                kingsType: table.kingsType,
                                kingsTypeId : table.kingsTypeId,
                                kingsName: table.kingsName,
                                top: table.top,
                                kings: {}
                            };

                                        newTable.kings[player.id] = player;

                                            name: player.name,

                                            picture: player.picture,
                                            statistic: player.statistic
                                        }
                                    } else {
                                        newTable.kings[player.id].statistic += player.statistic;
                                    }
                                }
                            });

                            tablesToShow.push(newTable);
                        });

                        return tablesToShow;
                    },

                        var $player = $playerTemplate.clone();

                                    .append($name)
                                    .appendTo($player.find(".player-data"));

                                players;

                            $ul.empty();

                            players = playersId
                                .sort(function (k1, k2) {
                                    return table.kings[k2].statistic - table.kings[k1].statistic;
                                })
                                .splice(0, table.top > 0 ? table.top : playersId.length);

                            if (players.length == 0) {
                                $ul
                                    .append("<li class=\"king-of\"><div class=\"player-data rtl center\">אין נתונים</div></li>")
                                    .data("FullTableLink", "");
                            } else {
                                players.forEach(function (playerId, index) {
                                    $ul.append(_buildPlayer(table.kings[playerId]));
                                });
                                $ul.data("FullTableLink", players.length < 4 ? "": moreURL.replace(/\{0\}/g, table.kingsTypeId).replace(/\{1\}/g, leagueFilter));
                            }
                        });

                        $(self).triggerHandler("KingsOf_refresh_view");
                    },

                    _init = function () {

                                .attr("href", "#" + table.kingsType)
                                .text(table.kingsName).end()
                                .appendTo($header);
                            $ul.attr("id", table.kingsType)
                                .appendTo($container)
                                .on("KingsOf_tab_active", _buildFullTableLink);

                            _refreshView();
                        });

                        $ddLeagues.change(function () {
                            leagueFilter = parseInt($ddLeagues.val(), 10);
                            _refreshView();
                        });
                    };

                _init()
            };

            return Module;
        })();
